📄 Leo Source Code

program private_token.aleo {

  record Token {
    owner: address,
    amount: u64,
  }

  transition mint(public amount: u64) -> Token {
    return Token { owner: self.caller, amount };
  }

  transition transfer(token: Token,
    public recipient: address,
    public amount: u64) -> (Token, Token) {
    assert(token.amount >= amount);
    return (
      Token { owner: token.owner, amount: token.amount - amount },
      Token { owner: recipient, amount }
    );
  }
}
